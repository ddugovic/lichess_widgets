Widgets for Lishogi

Add the following to the <head> of your webpage, and then add at least one of the widgets to the <body> of your webpage.

<link rel="stylesheet" href="https://fiss.mooo.com/lishogi_widgets/lishogi_widgets.css" />
<script src="https://fiss.mooo.com/lishogi_widgets/lishogi_widgets.js"></script>
<!-- For the light theme --> <link rel="stylesheet" href="https://fiss.mooo.com/lishogi_widgets/light.css" />
<!-- For the dark theme --> <link rel="stylesheet" href="https://fiss.mooo.com/lishogi_widgets/dark.css" />
<script>lishogi_widgets.profile("light", "Toadofsky", "Lishogi");</script>
<script>lishogi_widgets.profile("light", "Toadofsky");</script>
<script>lishogi_widgets.profile_scores("light", "Toadofsky");</script>
<script>lishogi_widgets.profile_scores("light", "Toadofsky", "");</script>
<script>lishogi_widgets.profile_big("light", "Toadofsky");</script>
<script>lishogi_widgets.profile_big("light", "Toadofsky", "My Lishogi");</script>
<script>lishogi_widgets.profile("dark", "Toadofsky", "Lishogi");</script>
<script>lishogi_widgets.profile("dark", "Toadofsky");</script>
<script>lishogi_widgets.profile_scores("dark", "Toadofsky");</script>
<script>lishogi_widgets.profile_scores("dark", "Toadofsky", "");</script>
<script>lishogi_widgets.profile_big("dark", "Toadofsky");</script>
<script>lishogi_widgets.profile_big("dark", "Toadofsky", "My Lishogi");</script>
